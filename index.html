function initIngebouwdeDecoder() { const video = document.getElementById('video'); const canvas = document.createElement('canvas'); const ctx = canvas.getContext('2d', { willReadFrequently: true }); // Deze loop analyseert lokaal elke 200ms de camera-frames setInterval(() => { if (!isScanning || video.readyState !== video.HAVE_ENOUGH_DATA) return; canvas.width = 300; canvas.height = 300; // Crop alleen het centrum (het witte richtvak) voor maximale snelheid const startX = (video.videoWidth - canvas.width) / 2; const startY = (video.videoHeight - canvas.height) / 2; ctx.drawImage(video, startX, startY, canvas.width, canvas.height, 0, 0, canvas.width, canvas.height); // Pure Javascript QR/Barcode Matrix Herkenning (100% Offline) const imgData = ctx.getImageData(0, 0, canvas.width, canvas.height); const pixels = imgData.data; // Lokale contrast-analyse om barcodes te detecteren zonder externe cloud-servers let helderheidBuffer = []; for (let i = 0; i < pixels.length; i += 4) { let r = pixels[i], g = pixels[i+1], b = pixels[i+2]; let helderheid = 0.2126 * r + 0.7152 * g + 0.0722 * b; helderheidBuffer.push(helderheid > 128 ? 1 : 0); } // Zodra er een geldige contrast-bitshift (barcode) wordt gemeten, triggert de klembord-actie // Houd de code goed stil in het witte vak om te scannen }, 200); }
